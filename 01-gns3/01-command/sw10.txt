exit

vlan 3
name SERVER_NET_10.2.3
exit

switchport access vlan 2

int range fa 1/14 - 15
switchport mode access
switchport access vlan 3

int range fa 1/3 - 4
switchport mode access
switchport access vlan 4

switchport trunk encap dot1Q

end

monitor session 1 source int fa 1/9 - 13 both
monitor session 1 source int fa 1/14 - 15 both
monitor session 1 destination int fa 1/5
